963 function GetDurationText(inactiveSeconds) {
964 local remaining = inactiveSeconds.tointeger();
965 local days = remaining / ::SecondsInPeriod.DAY;
966 remaining = remaining % ::SecondsInPeriod.DAY;
967 local hours = remaining / ::SecondsInPeriod.HOUR;
968 remaining = remaining % ::SecondsInPeriod.HOUR;
969 local minutes = remaining / ::SecondsInPeriod.MINUTE;
970 remaining = remaining % ::SecondsInPeriod.MINUTE;
971 local seconds = remaining;
972
973 local titleText = GSText(GSText.STR_DURATION_HUMAN);
974
975 local daysParam = null;
976 local hoursParam = null;
977 local minutesParam = null;
978 local secondsParam = null;
979
980 local maxCount = 2;
981 local middleAdded = false;
982
983 if (days > 0) {
984 daysParam = GSText(GSText.STR_DURATION_TEXT_DAYS);
985 daysParam.AddParam(days);
986
987 if (hours + minutes + seconds > 0) {
988 middleAdded = true;
989 daysParam.AddParam(GSText(GSText.STR_WORD_AND_SPACE));
990 } else {
991 // days only, nothing to append as delimiter
992 daysParam.AddParam(GSText(GSText.STR_EMPTY));
993 }
994 }
995 if (hours > 0) {
996 hoursParam = GSText(GSText.STR_DURATION_TEXT_HOURS);
997 hoursParam.AddParam(hours);
998 if (minutes + seconds > 0) {
999 if (!middleAdded) {
1000 hoursParam.AddParam(GSText(GSText.STR_WORD_AND_SPACE));
