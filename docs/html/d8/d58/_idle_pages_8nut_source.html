972 function GetDurationText(inactiveSeconds) {
973 local remaining = inactiveSeconds.tointeger();
974 local days = remaining / ::SecondsInPeriod.DAY;
975 remaining = remaining % ::SecondsInPeriod.DAY;
976 local hours = remaining / ::SecondsInPeriod.HOUR;
977 remaining = remaining % ::SecondsInPeriod.HOUR;
978 local minutes = remaining / ::SecondsInPeriod.MINUTE;
979 remaining = remaining % ::SecondsInPeriod.MINUTE;
980 local seconds = remaining;
981
982 local titleText = GSText(GSText.STR_DURATION_HUMAN);
983
984 local daysParam = null;
985 local hoursParam = null;
986 local minutesParam = null;
987 local secondsParam = null;
988
989 local maxCount = 2;
990 local middleAdded = false;
991
992 if (days > 0) {
993 daysParam = GSText(GSText.STR_DURATION_TEXT_DAYS);
994 daysParam.AddParam(days);
995
996 if (hours + minutes + seconds > 0) {
997 middleAdded = true;
998 daysParam.AddParam(GSText(GSText.STR_WORD_AND_SPACE));
999 } else {
1000 // days only, nothing to append as delimiter
1001 daysParam.AddParam(GSText(GSText.STR_EMPTY));
1002 }
1003 }
1004 if (hours > 0) {
1005 hoursParam = GSText(GSText.STR_DURATION_TEXT_HOURS);
