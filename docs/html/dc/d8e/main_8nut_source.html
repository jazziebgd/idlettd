158 function HandleEvents() {
159 local result = true;
160 while (GSEventController.IsEventWaiting()) {
161 local ev = GSEventController.GetNextEvent();
162 local ev_type = ev.GetEventType();
163 if (ev_type == GSEvent.ET_STORYPAGE_BUTTON_CLICK) {
164 local event = GSEventStoryPageButtonClick.Convert(ev);
165 local pageId = event.GetStoryPageID();
166 if (GSStoryPage.IsValidStoryPage(pageId) && pageId == this.IdleStoryInstance.GetPageID()) {
167 local buttonId = event.GetElementID();
168 this.IdleStoryInstance.HandleButtonClick(buttonId, event);
169 result = false;
170 };
171 }; else if (ev_type == GSEvent.ET_COMPANY_NEW) {
172 local company_event = GSEventCompanyNew.Convert(ev);
173 local company_id = company_event.GetCompanyID();
174 if (this.PlayerCompanyID == GSCompany.COMPANY_INVALID && company_id != GSCompany.COMPANY_INVALID) {
175 this.PlayerCompanyID = company_id;
176 result = false;
177 };
178 };
179 };
