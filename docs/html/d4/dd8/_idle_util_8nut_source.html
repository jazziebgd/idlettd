151 static function GetRunningCompanyVehicles(companyID, vehicleType = null, boolOnlyRunning = true) {
152 local items = GSList();
153 local vehicles = GSVehicleList();
154 local key = -1;
155 local resolvedID = GSCompany.ResolveCompanyID(companyID);
156 if (resolvedID == companyID) {
157 for (local vhc = vehicles.Begin(); !vehicles.IsEnd(); vhc = vehicles.Next()) {
158 if (
159 GSVehicle.IsValidVehicle(vhc)
160 && (vehicleType == null || GSVehicle.GetVehicleType(vhc) == vehicleType)
161 && GSVehicle.GetOwner(vhc) == resolvedID
162 && (!boolOnlyRunning || (GSVehicle.GetState(vhc) == GSVehicle.VS_RUNNING || GSVehicle.GetState(vhc) == GSVehicle.VS_AT_STATION))
163 ) {
164 key++;
165 items.AddItem(key, vhc);
166 }
167 }
168 }
169 return items;

197 if (!vehicles.IsEmpty()) {

202 local earnedLastYear = GSVehicle.GetProfitLastYear(vehicle);
203 vehiclesEarnedLastYear += earnedLastYear
204
205 }
206 vtData.count = vehicleCount;
207 vtData.balance = vehiclesEarnedLastYear;
208 vtData.idleBalance = (multiplier * vehiclesEarnedLastYear);
209 }
