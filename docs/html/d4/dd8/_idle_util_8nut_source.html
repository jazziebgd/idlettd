178 static function GetRunningCompanyVehicles(companyID, vehicleType = null, boolOnlyRunning = true) {
179 local items = GSList();
180 local vehicles = GSVehicleList();
181 local key = -1;
182 local resolvedID = GSCompany.ResolveCompanyID(companyID);
183 if (resolvedID == companyID) {
184 for (local vhc = vehicles.Begin(); !vehicles.IsEnd(); vhc = vehicles.Next()) {
185 if (
186 GSVehicle.IsValidVehicle(vhc)
187 && (vehicleType == null || GSVehicle.GetVehicleType(vhc) == vehicleType)
188 && GSVehicle.GetOwner(vhc) == resolvedID
189 && (!boolOnlyRunning || (GSVehicle.GetState(vhc) == GSVehicle.VS_RUNNING || GSVehicle.GetState(vhc) == GSVehicle.VS_AT_STATION))
190 ) {
191 key++;
192 items.AddItem(key, vhc);
193 }
194 }
195 }
196 return items;
