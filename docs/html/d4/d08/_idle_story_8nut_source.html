223 function Initialize(lastActiveTime, lastYearBalance, secondToLastYearBalance, intCurrentCash) {
224 this.IdlePagesInstance = IdlePages(this.PlayerCompanyID, this.IdleStoryPageID);
225 if (secondToLastYearBalance != 0 && lastYearBalance != 0) {
226 // set balance to 2nd last year first. It will be moved into place once last balance is set below
227 this.SetLastYearBalance(secondToLastYearBalance);
228 }
229 if (lastYearBalance != 0) {
230 this.SetLastYearBalance(lastYearBalance);
231 }
232 this.LastActiveTime = lastActiveTime;
233 this.CurrentCash = intCurrentCash;
234 if (lastActiveTime == 0) { // new game
235 this.SetLastInactiveSeconds(0);
236 this.LastSessionYearlyBalance = 0;
237 } else { // saved game
238 local currentTime = GSDate.GetSystemTime();
239 local inactiveSeconds = (currentTime - lastActiveTime).tointeger();
240 this.SetLastInactiveSeconds(inactiveSeconds);
241 this.LastSessionYearlyBalance = lastYearBalance;
242 }
243 if (!IdleUtil.HasHQ(this.PlayerCompanyID) && this.IdlePagesInstance != null) {
244 if (this.LastActiveTime == 0) { // new game
245 if (GSController.GetSetting("show_intro")) {
246 this.IdlePagesInstance.ShowIntroScreen();
247 }
248 } else { // loaded game without HQ built
249 this.IdlePagesInstance.ShowMissingHQScreen();
250 }
251 } else {
252 local idleBalance = this.CalculateIdleBalance();
253 this.ShowIdleReport(idleBalance);
254 }
255 Logger.Info("IdleTTD initialized - " + (lastActiveTime == 0 ? " (new game)." : " (loaded game)."));

266 function runIdleLoop() {
267 local date = GSDate.GetCurrentDate();
268 // Shortcuts.Process(this.PlayerCompanyID);
269 if (GSDate.IsValidDate(date)) {
270 local month = GSDate.GetMonth(date);
271 local day = GSDate.GetDayOfMonth(date);
272 if (month != this.PreviousMonth) {
273 if (this.PreviousMonth == 12) {
274 local newCash = this.GetCurrentCash();
275 local prevCash = this.CurrentCash;
276 this.CurrentCash = newCash;
277 local newBalance = this.CurrentCash - prevCash;
278 if (prevCash != 0) {
279 Logger.Table({newCash = newCash, prevCash = prevCash, newBalance = newBalance}, "Updated yearly balance", ::ScriptLogLevels.LOG_LEVEL_DEBUG);
280 this.SetLastYearBalance(newBalance);
281 if (this.IsAnyScreenOpen() && this.IdlePagesInstance.StatsScreenVisible) {
282 Logger.Verbose("Refreshing stats screen");
283 this.CacheVehicleStats(true);
284 this.IdlePagesInstance.ClearPage();
285 this.RenderStatsScreen();
286 }
287 }
288 }
289 this.PreviousMonth = month;
290 }

558 function CalculateIdleBalance() {
559 local idleMultiplier = IdleUtil.GetIdleMultiplier();
560 local currentTime = GSDate.GetSystemTime();
561 local inactiveSeconds = (currentTime - this.LastActiveTime).tointeger();
562 this.SetLastInactiveSeconds(inactiveSeconds);
563
564 // local passedDays = ((inactiveSeconds * 33.33) / 74).tointeger();
565 local passedDays = (floor(inactiveSeconds.tofloat() / ::SecondsPerGameDay.tofloat())).tointeger();
566 local idleBalance = (((this.LastYearBalance / 365.0) * passedDays) * idleMultiplier).tointeger();
567 return idleBalance;

751 function CacheVehicleStats(boolForce = false) {
752 if (this.PlayerCompanyID != GSCompany.COMPANY_INVALID && GSCompany.ResolveCompanyID(this.PlayerCompanyID) == this.PlayerCompanyID) {
753 local typeCount = IdleUtil.AllVehicleTypes.len();
754 if (boolForce == true || this._CachedAllVehicleStats.len() < typeCount) {
755 this._CachedAllVehicleStats = IdleUtil.GetAllVehicleStatsData(this.PlayerCompanyID);
756 }
757 } else {
758 this._CachedAllVehicleStats = [];
759 }
760 local totalVehicles = 0;
761 local totalVehiclesBalance = 0;
762 local totalVehiclesIdleBalance = 0;
763
764 foreach (index, vStats in this._CachedAllVehicleStats) {
765 totalVehicles += vStats.count;
766 totalVehiclesBalance += vStats.balance;
767 totalVehiclesIdleBalance += vStats.idleBalance;
768 }
769 this._CachedSummaryVehicleStats.count = totalVehicles;
770 this._CachedSummaryVehicleStats.balance = totalVehiclesBalance;
771 this._CachedSummaryVehicleStats.idleBalance = totalVehiclesIdleBalance;
