236 function Initialize(lastActiveTime, lastYearBalance, secondToLastYearBalance, intCurrentCash) {
237 this.IdlePagesInstance = IdlePages(this.PlayerCompanyID, this.IdleStoryPageID);
238 if (secondToLastYearBalance != 0 && lastYearBalance != 0) {
239 // set balance to 2nd last year first. It will be moved into place once last balance is set below
240 this.SetLastYearBalance(secondToLastYearBalance);
241 }
242 if (lastYearBalance != 0) {
243 this.SetLastYearBalance(lastYearBalance);
244 }
245 this._LastActiveTime = lastActiveTime;
246 this.CurrentCash = intCurrentCash;
247 if (lastActiveTime > 0) {
248 local currentTime = GSDate.GetSystemTime();
249 local inactiveSeconds = (currentTime - lastActiveTime).tointeger();
250 this.SetLastInactiveSeconds(inactiveSeconds);
251 this._LastSessionYearlyBalance = lastYearBalance;
252 } else {
253 this.SetLastInactiveSeconds(0);
254 this._LastSessionYearlyBalance = 0;
255 }
256 if (!IdleUtil.HasHQ(this.PlayerCompanyID) && this.IdlePagesInstance != null) {
257 if (this._LastActiveTime == 0) { // new game
258 if (GSController.GetSetting("show_intro")) {
259 this.IdlePagesInstance.ShowIntroScreen();
260 }
261 } else { // loaded game without HQ built
262 this.IdlePagesInstance.ShowMissingHQScreen();
263 }
264 } else {
265 local idleBalance = this.CalculateIdleBalance();
266 this.ShowIdleReport(idleBalance);
267 }
268 Logger.Info("IdleTTD initialized - " + (lastActiveTime == 0 ? " (new game)." : " (loaded game)."));

620 function CalculateIdleBalance() {
621 local idleMultiplier = IdleUtil.GetIdleMultiplier();
622 local currentTime = GSDate.GetSystemTime();
623 local inactiveSeconds = (currentTime - this._LastActiveTime).tointeger();
624 this.SetLastInactiveSeconds(inactiveSeconds);
625
626 // local passedDays = ((inactiveSeconds * 33.33) / 74).tointeger();
627 local passedDays = (floor(inactiveSeconds.tofloat() / ::SecondsPerGameDay.tofloat())).tointeger();
628 local idleBalance = (((this._LastYearBalance / 365.0) * passedDays) * idleMultiplier).tointeger();
629 return idleBalance;
630 }
