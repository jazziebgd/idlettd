222 function Initialize(lastActiveTime, lastYearBalance, secondToLastYearBalance, intCurrentCash) {
223 this.IdlePagesInstance = IdlePages(this.PlayerCompanyID, this.IdleStoryPageID);
224 if (secondToLastYearBalance != 0 && lastYearBalance != 0) {
225 // set balance to 2nd last year first. It will be moved into place once last balance is set below
226 this.SetLastYearBalance(secondToLastYearBalance);
227 }
228 if (lastYearBalance != 0) {
229 this.SetLastYearBalance(lastYearBalance);
230 }
231 this.LastActiveTime = lastActiveTime;
232 this.CurrentCash = intCurrentCash;
233 if (lastActiveTime == 0) { // new game
234 this.SetLastInactiveSeconds(0);
235 this.LastSessionYearlyBalance = 0;
236 } else { // saved game
237 local currentTime = GSDate.GetSystemTime();
238 local inactiveSeconds = (currentTime - lastActiveTime).tointeger();
239 this.SetLastInactiveSeconds(inactiveSeconds);
240 this.LastSessionYearlyBalance = lastYearBalance;
241 }
242 if (!IdleUtil.HasHQ(this.PlayerCompanyID) && this.IdlePagesInstance != null) {
243 if (this.LastActiveTime == 0) { // new game
244 if (GSController.GetSetting("show_intro")) {
245 this.IdlePagesInstance.ShowIntroScreen();
246 } else {
247 this.IdlePagesInstance.RenderMissingHQScreen();
248 }
249 } else { // loaded game without HQ built
250 this.IdlePagesInstance.ShowMissingHQScreen();
251 }
252 } else {
253 local idleBalance = this.CalculateIdleBalance();
254 this.ShowIdleReport(idleBalance);
255 }
256 Logger.Info("IdleTTD initialized - " + (lastActiveTime == 0 ? " (new game)." : " (loaded game)."));
257 }

267 function runIdleLoop() {
268 local date = GSDate.GetCurrentDate();
269 if (GSDate.IsValidDate(date)) {
270 local month = GSDate.GetMonth(date);
271 local day = GSDate.GetDayOfMonth(date);
272 local year = GSDate.GetYear(date);
273 if (month != this.PreviousMonth) {
274 if (this.PreviousMonth == 12) {
275 local newCash = this.GetCurrentCash();
276 local prevCash = this.CurrentCash;
277 this.CurrentCash = newCash;
278 local newBalance = this.CurrentCash - prevCash;
279 if (prevCash != 0) {
280 Logger.Table({prevCash = prevCash, newCash = newCash, newBalance = newBalance}, "Year " + (year - 1) + " balance info", ::ScriptLogLevels.LOG_LEVEL_DEBUG);
281 this.SetLastYearBalance(newBalance);
282 if (this.IdlePagesInstance.StatsScreenVisible) {
283 Logger.Verbose("Refreshing stats screen");
284 this.CacheVehicleStats(true);
285 this.IdlePagesInstance.ClearPage();
286 this.RenderStatsScreen();
287 }
288 }
289 }
290 this.PreviousMonth = month;
291 }

559 function CalculateIdleBalance() {
560 local idleMultiplier = IdleUtil.GetIdleMultiplier();
561 local currentTime = GSDate.GetSystemTime();
562 local inactiveSeconds = (currentTime - this.LastActiveTime).tointeger();
563 this.SetLastInactiveSeconds(inactiveSeconds);
564
565 local passedDays = (floor(inactiveSeconds.tofloat() / ::SecondsPerGameDay.tofloat())).tointeger();
566 local idleBalance = (((this.LastYearBalance / 365.0) * passedDays) * idleMultiplier).tointeger();
567 return idleBalance;
568 }

751 function CacheVehicleStats(boolForce = false) {
752 if (this.PlayerCompanyID != GSCompany.COMPANY_INVALID && GSCompany.ResolveCompanyID(this.PlayerCompanyID) == this.PlayerCompanyID) {
753 local typeCount = IdleUtil.AllVehicleTypes.len();
754 if (boolForce == true || this._CachedAllVehicleStats.len() < typeCount) {
755 this._CachedAllVehicleStats = IdleUtil.GetAllVehicleStatsData(this.PlayerCompanyID);
756 }
757 } else {
758 this._CachedAllVehicleStats = [];
759 }
760 local totalVehicles = 0;
761 local totalVehiclesBalance = 0;
762 local totalVehiclesIdleBalance = 0;
763
764 foreach (index, vStats in this._CachedAllVehicleStats) {
765 totalVehicles += vStats.count;
766 totalVehiclesBalance += vStats.balance;
767 totalVehiclesIdleBalance += vStats.idleBalance;
768 }
769 this._CachedSummaryVehicleStats.count = totalVehicles;
770 this._CachedSummaryVehicleStats.balance = totalVehiclesBalance;
771 this._CachedSummaryVehicleStats.idleBalance = totalVehiclesIdleBalance;
772 if (totalVehicles > 0) {
