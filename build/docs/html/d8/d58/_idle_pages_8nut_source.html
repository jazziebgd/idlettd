388 /**
389 @brief Shows idle report screen
390
391 @details Clears idle story page, renders idle report elements with [this.RenderIdleReportScreen()](#RenderIdleReportScreen) and shows the page to the player. For details on idle report contents check out #RenderIdleReportScreen function.
392
393 @note This will set #IdleReportVisible flag to `true`, causing [MainClass.Start()](#MainClass.Start) `while` loop to run a lot more often than normal. It might slow the game down but ensures that script react promptly to clicks and other player actions.
394
395 @attention Idle report page is \em quasi-persistent. If player closes story book without clicking the button within, script will open idle report story book page again and keep doing it until player clicks the button on the page.
396
397 @param idleBalance Idle balance to use
398 @param inactiveSeconds Inactive idle seconds
399 @param vehicleSummary Summary of all vehicles
400 @param allVehicleStats An array of vehicle stats for all vehicle types
401 @param intLastYearBalance Last year balance for calculations
402
403 @return bool
404 @retval true Idle report story book page is displayed
405 @retval false Failed opening idle report story book page
406 */
407 function ShowIdleReportScreen(idleBalance = 0, inactiveSeconds = 0, vehicleSummary = null, allVehicleStats = null, intLastYearBalance = 0) {
408 this.ClearPage();
409 local hasHQ = IdleUtil.HasHQ(this.PlayerCompanyID);
410 if (hasHQ) {
411 if (idleBalance == 0) {
412 return this.ShowStatsScreen(vehicleSummary, allVehicleStats, idleBalance, inactiveSeconds, intLastYearBalance);
413 }
414 this.RenderIdleReportScreen(idleBalance, inactiveSeconds, vehicleSummary, allVehicleStats);
415 this._RenderDebugNavButtons();
416 return IdleUtil.DisplayPage(this.IdleStoryPageID);
417 }
418 return false;
419 }

1051 @param inactiveSeconds Total seconds to show duration for
1052
1053 @returns GSText Text element with formatted duration

1055 function GetDurationText(inactiveSeconds) {
1056 local remaining = inactiveSeconds.tointeger();
1057 local days = remaining / ::SecondsInPeriod.DAY;
1058 remaining = remaining % ::SecondsInPeriod.DAY;
1059 local hours = remaining / ::SecondsInPeriod.HOUR;
1060 remaining = remaining % ::SecondsInPeriod.HOUR;
1061 local minutes = remaining / ::SecondsInPeriod.MINUTE;
1062 remaining = remaining % ::SecondsInPeriod.MINUTE;
1063 local seconds = remaining;
1064
1065 local titleText = GSText(GSText.STR_DURATION_TEXT);
1066
1067 if (days > 0) {
1068 titleText.AddParam(GSText(GSText.STR_NUMBER_OF_DAYS, days));
1069 if (hours + minutes + seconds > 0) {
1070 // append delimiter
1071 if (minutes + seconds == 0) {
1072 // hours are last, append "and" here as delimiter
1073 titleText.AddParam(GSText(GSText.STR_WORD_AND_SPACE));
1074 } else {
1075 // either minutes or seconds follow, append comma here as delimiter
1076 titleText.AddParam(GSText(GSText.STR_COMMA_SPACE));
1077 }
1078 } else {
1079 // days only, nothing to append as delimiter
1080 titleText.AddParam(GSText(GSText.STR_EMPTY));
1081 }
1082 } else {
1083 // no days at all
1084 titleText.AddParam(GSText(GSText.STR_EMPTY, 0));
1085 // no delimiter to append
1086 titleText.AddParam(GSText(GSText.STR_EMPTY));
1087 }
1088 if (hours > 0) {
1089 titleText.AddParam(GSText(GSText.STR_NUMBER_OF_HOURS, hours));

1091 // append delimiter
1092 if (seconds == 0) {
1093 // minutes are last, append "and" here as delimiter
1094 titleText.AddParam(GSText(GSText.STR_WORD_AND_SPACE));

1098 }

1100 // hours are last, nothing left to add
1101 titleText.AddParam(GSText(GSText.STR_EMPTY));
1102 }
1103 } else {
1104 // no hours at all
1105 titleText.AddParam(GSText(GSText.STR_EMPTY, 0));

1107 titleText.AddParam(GSText(GSText.STR_EMPTY));
1108 }
1109

1113 titleText.AddParam(GSText(GSText.STR_WORD_AND_SPACE));

1171 @param totalSeconds Number of real-time seconds passed in period od time
1172 @param forceTimeUnit Force time unit index (1 = minute, 2 = hour, 3 = day)
1173 @param intMinTreshold Minimum treshold - values below this will cause next (longer) time unit to be picked
1174
1175 @returns int Id of balance text element
1176 */
1177 function GetBalancePerTimeUnit(totalAmount, totalSeconds, forceTimeUnit = 0, intMinTreshold = 1) {
1178 local balanceInTimeUnitText = GSText(GSText.STR_EMPTY);
1179 local amount = 0;
1180 local newAmount = 0;
1181 local timeUnitString = GSText(GSText.STR_EMPTY);
1182 local unitPicked = 0;

1187 newAmount = perSecond;
1188 timeUnitString = GSText(GSText.STR_PER_SECOND);
1189 if (forceTimeUnit <= 0) {
1190 if ((totalAmount > 0 && newAmount < intMinTreshold) || (totalAmount < 0 && newAmount > -1 * intMinTreshold)) {
1191 newAmount = perSecond * ::SecondsInPeriod.MINUTE;
1192 timeUnitString = GSText(GSText.STR_PER_MINUTE);
1193 unitPicked = 1;
1194 }
1195
1196 if ((totalAmount > 0 && newAmount < intMinTreshold) || (totalAmount < 0 && newAmount > -1 * intMinTreshold)) {
1197 newAmount = perSecond * ::SecondsInPeriod.HOUR;
1198 timeUnitString = GSText(GSText.STR_PER_HOUR);
1199 unitPicked = 2;
1200 }
1201
